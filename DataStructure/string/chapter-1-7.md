```
class Solution {
public:
    int uniqueMorseRepresentations(vector<string>& words) {
        // 哈希表法
        int len = words.size();
        if(len == 0){
            return 0;
        }
        vector<string> datas = {".-","-...","-.-.","-..",".","..-.","--.","....","..",".---","-.-",".-..","--","-.","---",".--.","--.-",".-.","...","-","..-","...-",".--","-..-","-.--","--.."};
        unordered_set<string> tag;
        for(int i = 0; i < len; i ++){
            string str;
            for(int j = 0; j < words[i].size(); j ++){
                char ch = words[i][j];
                if(ch >= 'a' && ch <= 'z'){
                    str += datas[ch - 'a'];
                }else if(ch >= 'A' && ch <= 'Z'){
                    str += datas[ch - 'A'];
                }else{
                    return -1;
                }
            }
            tag.insert(str);
        }
        return tag.size();
    }
};
```
